The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

from zipfile import ZipFile
from PIL import Image, ImageDraw, ImageFont
import pytesseract
import cv2 as cv
import numpy as np
from IPython.display import display
import gc



# loading the face detection classifier
face_cascade = cv.CascadeClassifier('readonly/haarcascade_frontalface_default.xml')


# FUNCTIONS

# Main
def launch_photo_finder (zipfilename, word, adjcoef, neighbors):
    global files_dict
    files_dict={}
    with ZipFile(zipfilename, 'r') as zip:
        for f in zip.namelist():
            with zip.open(f) as file:
                img=Image.open(file)
                files_dict[f]=img.convert('RGB') 

            
    for file in files_dict.keys():
        if word_searcher(file, word):
            print("Results found in file {}".format(file))
            search_photos = photo_searcher(file, adjcoef, neighbors)
            if len(search_photos)==0:
                print("But there were no faces in that file!")
            else:
                display(create_sheet(search_photos))
    gc.collect()

# Extract text from file
def word_searcher(file, word):
    img_l = files_dict[file].convert('L')
    text = pytesseract.image_to_string(img_l) 
    text.replace('-\n','')
    if text.find(word)>1:
        return True
    else:
        return False

# Construct list of photos
def photo_searcher(file, adjcoef, neighbors):
    image_array = np.array(files_dict[file])
    faces = face_cascade.detectMultiScale(image_array, adjcoef, neighbors)
    
    images=[]
    for face in faces:
        face_pic=files_dict[file].crop((face[0],face[1], face[0]+face[2],face[1]+face[3]))
        face_pic.thumbnail((100, 100))
        images.append(face_pic)
    
    return images

# Calculate N rows for contact sheet
def row_calc(length_vector):
    val=int(length_vector/5+1)
    if length_vector%5==0:
        val=val-1
    return val

#Create Contact Sheet
def create_sheet(images):
    contact_sheet=Image.new(images[0].mode, (100*5, row_calc(len(images))*100))
    extended_sheet = ImageDraw.Draw(contact_sheet)
    x=0
    y=0
    for img in images:
        contact_sheet.paste(img, (x, y))
        if x+100 == contact_sheet.width:
            x=0
            y=y+100
        else:
            x=x+100
    return contact_sheet
